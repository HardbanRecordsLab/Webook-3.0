3. MANIFEST SERVICE (manifestService.ts)
typescript
import Ajv from 'ajv';
import manifestSchema from '../schemas/manifest.schema.json';

const ajv = new Ajv();
const validateSchema = ajv.compile(manifestSchema);

export class ManifestService {
  
  // Parse i walidacja manifestu
  static async parseManifest(file: File): Promise<AssemblyManifest> {
    const text = await file.text();
    const data = JSON.parse(text);
    
    if (!validateSchema(data)) {
      throw new Error(`Invalid manifest: ${JSON.stringify(validateSchema.errors)}`);
    }
    
    return data as AssemblyManifest;
  }
  
  // Ekstrakcja wszystkich placeholderów
  static extractPlaceholders(manifest: AssemblyManifest): PlaceholderMapping[] {
    const mappings: PlaceholderMapping[] = [];
    
    manifest.sections.forEach(section => {
      Object.entries(section.placeholders).forEach(([key, path]) => {
        mappings.push({
          placeholder: key,
          type: this.detectAssetType(path),
          sourcePath: path,
          resolved: false
        });
      });
    });
    
    return mappings;
  }
  
  // Detekcja typu assetu z rozszerzenia
  private static detectAssetType(path: string): AssetType {
    const ext = path.split('.').pop()?.toLowerCase();
    
    if (['html', 'htm'].includes(ext!)) return 'workcard-html';
    if (['mp3', 'wav', 'ogg'].includes(ext!)) return 'audio';
    if (['jpg', 'jpeg', 'png', 'webp'].includes(ext!)) return 'raster';
    if (['svg'].includes(ext!)) return 'vector';
    
    return 'source';
  }
  
  // Walidacja struktury (bez AI)
  static validateStructure(manifest: AssemblyManifest): ValidationError[] {
    const errors: ValidationError[] = [];
    
    // Check sections order
    const chapters = manifest.sections.map(s => s.chapter);
    if (chapters.some((ch, i) => i > 0 && ch <= chapters[i-1])) {
      errors.push({
        type: 'INVALID_PATH',
        placeholder: 'sections',
        message: 'Chapters must be in ascending order',
        severity: 'ERROR'
      });
    }
    
    // Check duplicate placeholders
    const allPlaceholders = manifest.sections.flatMap(s => 
      Object.keys(s.placeholders)
    );
    const duplicates = allPlaceholders.filter((p, i) => 
      allPlaceholders.indexOf(p) !== i
    );
    
    if (duplicates.length > 0) {
      errors.push({
        type: 'INVALID_PATH',
        placeholder: duplicates[0],
        message: `Duplicate placeholder: ${duplicates[0]}`,
        severity: 'ERROR'
      });
    }
    
    return errors;
  }
}
4. PLACEHOLDER PARSER (placeholderParser.ts)
typescript
export class PlaceholderParser {
  
  // Wykryj wszystkie {{PLACEHOLDERS}} w tekście
  static detectInText(text: string): string[] {
    const regex = /\{\{([A-Z0-9_]+)\}\}/g;
    const matches = [...text.matchAll(regex)];
    return [...new Set(matches.map(m => `{{${m[1]}}}`))];
  }
  
  // Zamień placeholdery na rzeczywisty content
  static async injectContent(
    text: string,
    mappings: Map<string, ProcessedAsset>
  ): Promise<string> {
    let result = text;
    
    for (const [placeholder, asset] of mappings) {
      const replacement = await this.generateReplacement(asset);
      result = result.replaceAll(placeholder, replacement);
    }
    
    return result;
  }
  
  // Generuj HTML replacement w zależności od typu
  private static async generateReplacement(asset: ProcessedAsset): Promise<string> {
    switch (asset.type) {
      case 'workcard-html':
        return `
          <section 
            data-workcard-id="${asset.id}"
            data-checksum="${asset.checksum}"
            class="workcard-inject"
            style="isolation: isolate;"
          >
            ${await fetch(asset.processedURL).then(r => r.text())}
          </section>
        `;
      
      case 'audio':
        return `
          <audio 
            controls 
            data-asset-id="${asset.id}"
            class="audio-inject"
          >
            <source src="${asset.processedURL}" type="audio/mpeg">
          </audio>
        `;
      
      case 'raster':
        return `
          <figure data-asset-id="${asset.id}" class="image-inject">
            <img 
              src="${asset.processedURL}" 
              alt="Asset ${asset.id}"
              loading="lazy"
            />
          </figure>
        `;
      
      default:
        return `<!-- UNSUPPORTED ASSET TYPE: ${asset.type} -->`;
    }
  }
}

export class AssemblyValidator {
  
  // AI-powered semantic validation (TYLKO validation, zero content generation)
  static async validateWithAI(
    manifest: AssemblyManifest,
    mappings: PlaceholderMapping[]
  ): Promise<ValidationResult> {
    
    const prompt = `
STRICT VALIDATION TASK - DO NOT GENERATE CONTENT

Manifest sections: ${manifest.sections.length}
Expected placeholders: ${mappings.length}

Placeholders status:
${mappings.map(m => `${m.placeholder}: ${m.resolved ? 'RESOLVED' : 'MISSING'}`).join('\n')}

YOUR TASK:
1. Identify missing assets
2. Detect orphaned files
3. Check naming consistency
4. Report errors

RETURN ONLY JSON:
{
  "status": "COMPLETE" | "INCOMPLETE",
  "missing": ["{{PLACEHOLDER}}", ...],
  "orphaned": ["filename.ext", ...],
  "warnings": ["message", ...]
}

DO NOT:
- Generate placeholder content
- Suggest layouts
- Modify manifest
- Create assets
`;

    const response = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1000,
        messages: [{ role: 'user', content: prompt }]
      })
    });
    
    const data = await response.json();
    const result = JSON.parse(
      data.content[0].text.replace(/```json|```/g, '').trim()
    );
    
    return {
      status: result.status,
      total: mappings.length,
      mapped: mappings.filter(m => m.resolved).length,
      missing: mappings.filter(m => !m.resolved),
      orphaned: result.orphaned || [],
      errors: result.warnings?.map((w: string) => ({
        type: 'MISSING_ASSET',
        placeholder: '',
        message: w,
        severity: 'WARNING'
      })) || []
    };
  }
  
  // Lokalna walidacja (bez AI)
  static validateLocal(mappings: PlaceholderMapping[]): ValidationResult {
    return {
      status: mappings.every(m => m.resolved) ? 'COMPLETE' : 'INCOMPLETE',
      total: mappings.length,
      mapped: mappings.filter(m => m.resolved).length,
      missing: mappings.filter(m => !m.resolved),
      orphaned: [],
      errors: []
    };
  }
}
6. EXPORT ENGINE (exportEngine.ts)

export class ExportEngine {
  
  // Export do HTML (single file)
  static async exportHTML(
    manifest: AssemblyManifest,
    assembledContent: string,
    assets: ProcessedAsset[]
  ): Promise<ExportResult> {
    
    const html = `
<!DOCTYPE html>
<html lang="${manifest.metadata.language}">
<head>
  <meta charset="UTF-8">
  <meta name="author" content="${manifest.metadata.author}">
  <title>${manifest.metadata.title}</title>
  <style>${await this.bundleCSS(manifest.assets.css || [])}</style>
</head>
<body>
  <main class="ebook-container">
    ${assembledContent}
  </main>
  <script>${await this.bundleScripts(manifest.assets.scripts || [])}</script>
</body>
</html>
    `;
    
    const blob = new Blob([html], { type: 'text/html' });
    const url = URL.createObjectURL(blob);
    
    return {
      success: true,
      format: 'html',
      size: blob.size,
      downloadURL: url,
      metadata: {
        assets: assets.length,
        generatedAt: new Date().toISOString()
      }
    };
  }
  
  // Export do EPUB (via worker)
  static async exportEPUB(
    manifest: AssemblyManifest,
    assembledContent: string,
    assets: ProcessedAsset[]
  ): Promise<ExportResult> {
    
    const worker = new Worker(new URL('../workers/epub.worker.ts', import.meta.url));
    
    return new Promise((resolve, reject) => {
      worker.postMessage({
        manifest,
        content: assembledContent,
        assets
      });
      
      worker.onmessage = (e) => {
        if (e.data.success) {
          resolve(e.data);
        } else {
          reject(e.data.error);
        }
        worker.terminate();
      };
    });
  }
  
  // Export do PDF (via worker - heavy processing)
  static async exportPDF(
    manifest: AssemblyManifest,
    assembledContent: string,
    config: ExportConfig
  ): Promise<ExportResult> {
    
    const worker = new Worker(new URL('../workers/pdf.worker.ts', import.meta.url));
    
    return new Promise((resolve, reject) => {
      worker.postMessage({
        manifest,
        content: assembledContent,
        config
      });
      
      worker.onmessage = (e) => {
        if (e.data.success) {
          resolve(e.data);
        } else {
          reject(e.data.error);
        }
        worker.terminate();
      };
    });
  }
  
  private static async bundleCSS(cssPaths: string[]): Promise<string> {
    const styles = await Promise.all(
      cssPaths.map(path => fetch(path).then(r => r.text()))
    );
    return styles.join('\n\n');
  }
  
  private static async bundleScripts(scriptPaths: string[]): Promise<string> {
    const scripts = await Promise.all(
      scriptPaths.map(path => fetch(path).then(r => r.text()))
    );
    return scripts.join('\n\n');
  }
}

// services/assembly/htmlSanitizer.ts

import DOMPurify from 'dompurify';

export class HTMLSanitizer {
  
  private static ALLOWED_TAGS = [
    'div', 'span', 'p', 'h1', 'h2', 'h3', 'h4', 'h5', 'h6',
    'ul', 'ol', 'li', 'table', 'tr', 'td', 'th',
    'input', 'textarea', 'label', 'button', 'select', 'option',
    'img', 'audio', 'video', 'canvas'
  ];
  
  private static ALLOWED_ATTRS = [
    'class', 'id', 'style', 'data-*', 
    'type', 'value', 'placeholder', 'name',
    'src', 'alt', 'width', 'height'
  ];
  
  private static FORBIDDEN_PATTERNS = [
    /javascript:/gi,
    /on\w+=/gi,  // onclick, onerror, etc.
    /<script/gi,
    /<iframe/gi,
    /eval\(/gi
  ];
  
  // Sanitize HTML przed injection
  static sanitize(html: string, strict: boolean = true): string {
    
    // 1. Check for forbidden patterns
    for (const pattern of this.FORBIDDEN_PATTERNS) {
      if (pattern.test(html)) {
        throw new AssemblyError(
          'SECURITY_VIOLATION',
          'ERROR',
          `Forbidden pattern detected: ${pattern.source}`
        );
      }
    }
    
    // 2. DOMPurify clean
    const clean = DOMPurify.sanitize(html, {
      ALLOWED_TAGS: strict ? this.ALLOWED_TAGS : undefined,
      ALLOWED_ATTR: this.ALLOWED_ATTRS,
      ALLOW_DATA_ATTR: true,
      FORBID_TAGS: ['script', 'iframe', 'object', 'embed'],
      FORBID_ATTR: ['onerror', 'onload', 'onclick']
    });
    
    // 3. Wrap w sandboxed container
    return `
      <div 
        class="workcard-sandbox" 
        data-sanitized="true"
        style="
          all: initial;
          display: block;
          isolation: isolate;
          contain: layout style paint;
        "
      >
        ${clean}
      </div>
    `;
  }
  
  // CSS isolation via Shadow DOM (optional)
  static wrapInShadowDOM(html: string, css: string): string {
    return `
      <div class="shadow-host">
        <template shadowrootmode="open">
          <style>
            :host { all: initial; display: block; }
            ${css}
          </style>
          ${html}
        </template>
      </div>
    `;
  }
}

// services/assembly/assetProcessor.ts

export class AssetProcessor {
  
  // Process uploaded file → ProcessedAsset
  static async process(
    file: File,
    type: AssetType,
    config: ProcessingConfig
  ): Promise<ProcessedAsset> {
    
    switch (type) {
      case 'workcard-html':
        return this.processHTML(file, config);
      case 'raster':
        return this.processImage(file, config);
      case 'audio':
        return this.processAudio(file, config);
      default:
        return this.processGeneric(file);
    }
  }
  
  // HTML processing
  private static async processHTML(
    file: File, 
    config: ProcessingConfig
  ): Promise<ProcessedAsset> {
    
    const html = await file.text();
    
    // Extract inline CSS
    const cssMatch = html.match(/<style[^>]*>([\s\S]*?)<\/style>/gi);
    const inlineCSS = cssMatch ? cssMatch.join('\n') : '';
    
    // Remove <html>, <head>, <body> wrappers
    const bodyContent = html
      .replace(/<\/?html[^>]*>/gi, '')
      .replace(/<head[^>]*>[\s\S]*?<\/head>/gi, '')
      .replace(/<\/?body[^>]*>/gi, '');
    
    // Sanitize
    const sanitized = HTMLSanitizer.sanitize(bodyContent);
    
    // Generate blob
    const blob = new Blob([sanitized], { type: 'text/html' });
    const url = URL.createObjectURL(blob);
    
    return {
      id: this.generateAssetId(),
      originalPath: file.name,
      processedURL: url,
      type: 'workcard-html',
      size: blob.size,
      checksum: await this.generateChecksum(blob),
      metadata: {
        inlineCSS,
        sanitized: true
      }
    };
  }
  
  // Image optimization
  private static async processImage(
    file: File,
    config: ProcessingConfig
  ): Promise<ProcessedAsset> {
    
    const img = await createImageBitmap(file);
    
    const canvas = new OffscreenCanvas(
      config.maxWidth || img.width,
      config.maxHeight || img.height
    );
    
    const ctx = canvas.getContext('2d')!;
    
    // Resize preserving aspect ratio
    const ratio = Math.min(
      canvas.width / img.width,
      canvas.height / img.height
    );
    
    const w = img.width * ratio;
    const h = img.height * ratio;
    
    ctx.drawImage(img, 0, 0, w, h);
    
    // Convert to blob
    const blob = await canvas.convertToBlob({
      type: config.format || 'image/webp',
      quality: config.quality || 0.85
    });
    
    const url = URL.createObjectURL(blob);
    
    return {
      id: this.generateAssetId(),
      originalPath: file.name,
      processedURL: url,
      type: 'raster',
      size: blob.size,
      checksum: await this.generateChecksum(blob),
      metadata: {
        originalSize: file.size,
        compressionRatio: ((1 - blob.size / file.size) * 100).toFixed(1) + '%',
        dimensions: { width: w, height: h }
      }
    };
  }
  
  // Audio normalization
  private static async processAudio(
    file: File,
    config: ProcessingConfig
  ): Promise<ProcessedAsset> {
    
    // Dla audio możemy tylko validate + convert format
    const audioContext = new AudioContext();
    const arrayBuffer = await file.arrayBuffer();
    const audioBuffer = await audioContext.decodeAudioData(arrayBuffer);
    
    // Optional: normalize volume, trim silence, etc.
    // For now just pass-through
    
    const url = URL.createObjectURL(file);
    
    return {
      id: this.generateAssetId(),
      originalPath: file.name,
      processedURL: url,
      type: 'audio',
      size: file.size,
      checksum: await this.generateChecksum(file),
      metadata: {
        duration: audioBuffer.duration,
        sampleRate: audioBuffer.sampleRate,
        channels: audioBuffer.numberOfChannels
      }
    };
  }
  
  // Checksum generation (integrity verification)
  private static async generateChecksum(blob: Blob): Promise<string> {
    const buffer = await blob.arrayBuffer();
    const hashBuffer = await crypto.subtle.digest('SHA-256', buffer);
    const hashArray = Array.from(new Uint8Array(hashBuffer));
    return hashArray.map(b => b.toString(16).padStart(2, '0')).join('');
  }
  
  private static generateAssetId(): string {
    return `asset_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
  }
}

interface ProcessingConfig {
  maxWidth?: number;
  maxHeight?: number;
  quality?: number;
  format?: 'image/webp' | 'image/jpeg' | 'image/png';
}

// services/assembly/incrementalValidator.ts

export class IncrementalValidator {
  
  private mappings = new Map<string, PlaceholderMapping>();
  private listeners: Array<(result: PartialValidationResult) => void> = [];
  
  // Subscribe to validation updates
  onUpdate(callback: (result: PartialValidationResult) => void) {
    this.listeners.push(callback);
  }
  
  // Add/update single mapping
  async updateMapping(placeholder: string, asset: ProcessedAsset | null) {
    
    const mapping = this.mappings.get(placeholder);
    if (!mapping) return;
    
    mapping.asset = asset;
    mapping.resolved = asset !== null;
    
    // Instant validation
    const validation = this.validateSingle(mapping);
    
    // Notify listeners
    this.notifyUpdate({
      placeholder,
      status: validation.status,
      errors: validation.errors,
      progress: this.calculateProgress()
    });
  }
  
  // Validate single mapping
  private validateSingle(mapping: PlaceholderMapping): ValidationResult {
    const errors: ValidationError[] = [];
    
    if (!mapping.resolved) {
      errors.push({
        type: 'MISSING_ASSET',
        placeholder: mapping.placeholder,
        message: `Asset not mapped for ${mapping.placeholder}`,
        severity: 'ERROR'
      });
    }
    
    if (mapping.asset) {
      // Type mismatch check
      const expectedType = this.inferExpectedType(mapping.placeholder);
      if (expectedType && mapping.asset.type !== expectedType) {
        errors.push({
          type: 'UNSUPPORTED_TYPE',
          placeholder: mapping.placeholder,
          message: `Expected ${expectedType}, got ${mapping.asset.type}`,
          severity: 'WARNING'
        });
      }
      
      // Size check
      if (mapping.asset.size > 10 * 1024 * 1024) { // 10MB
        errors.push({
          type: 'INVALID_PATH',
          placeholder: mapping.placeholder,
          message: `Asset size ${(mapping.asset.size / 1024 / 1024).toFixed(1)}MB exceeds 10MB limit`,
          severity: 'WARNING'
        });
      }
    }
    
    return {
      status: errors.some(e => e.severity === 'ERROR') ? 'INCOMPLETE' : 'COMPLETE',
      total: 1,
      mapped: mapping.resolved ? 1 : 0,
      missing: mapping.resolved ? [] : [mapping],
      orphaned: [],
      errors
    };
  }
  
  // Infer expected type from placeholder name
  private inferExpectedType(placeholder: string): AssetType | null {
    if (/WORKCARD|CARD|WORKSHEET/i.test(placeholder)) return 'workcard-html';
    if (/AUDIO|SOUND|MUSIC/i.test(placeholder)) return 'audio';
    if (/IMAGE|IMG|PHOTO|PICTURE/i.test(placeholder)) return 'raster';
    return null;
  }
  
  // Calculate overall progress
  private calculateProgress(): number {
    const total = this.mappings.size;
    const resolved = Array.from(this.mappings.values()).filter(m => m.resolved).length;
    return total > 0 ? (resolved / total) * 100 : 0;
  }
  
  private notifyUpdate(result: PartialValidationResult) {
    this.listeners.forEach(cb => cb(result));
  }
}

interface PartialValidationResult {
  placeholder: string;
  status: 'COMPLETE' | 'INCOMPLETE';
  errors: ValidationError[];
  progress: number;
}

// services/assembly/commandStack.ts

interface Command {
  execute(): Promise<void>;
  undo(): Promise<void>;
  description: string;
}

export class CommandStack {
  
  private stack: Command[] = [];
  private pointer: number = -1;
  
  async execute(command: Command) {
    // Remove future commands if we're in the middle of history
    this.stack = this.stack.slice(0, this.pointer + 1);
    
    await command.execute();
    
    this.stack.push(command);
    this.pointer++;
  }
  
  async undo() {
    if (!this.canUndo()) return;
    
    await this.stack[this.pointer].undo();
    this.pointer--;
  }
  
  async redo() {
    if (!this.canRedo()) return;
    
    this.pointer++;
    await this.stack[this.pointer].execute();
  }
  
  canUndo(): boolean {
    return this.pointer >= 0;
  }
  
  canRedo(): boolean {
    return this.pointer < this.stack.length - 1;
  }
  
  getHistory(): string[] {
    return this.stack.map(c => c.description);
  }
}

// Example: MapAssetCommand
export class MapAssetCommand implements Command {

  }

// services/assembly/batchMapper.ts

export class BatchMapper {
  
  // Auto-map by naming convention
  static autoMapByNaming(
    placeholders: PlaceholderMapping[],
    assets: File[]
  ): Map<string, File> {
    
    const mappings = new Map<string, File>();
    
    for (const placeholder of placeholders) {
      // Extract semantic parts
      // {{WORKCARD_DAY_01}} → "workcard", "day", "01"
      const parts = this.parsePlaceholder(placeholder.placeholder);
      
      // Find best matching file
      const match = assets.find(file => {
        const fileName = file.name.toLowerCase();
        return parts.every(part => fileName.includes(part));
      });
      
      if (match) {
        mappings.set(placeholder.placeholder, match);
      }
    }
    
    return mappings;
  }
  
  // Bulk upload folder
  static async uploadFolder(
    folderInput: HTMLInputElement
  ): Promise<Map<string, File>> {
    
    const files = Array.from(folderInput.files || []);
    const organized = new Map<string, File>();
    
    files.forEach(file => {
      // Preserve folder structure
      const path = file.webkitRelativePath || file.name;
      organized.set(path, file);
    });
    
    return organized;
  }
  
  // Pattern-based mapping
  static mapByPattern(
    placeholders: PlaceholderMapping[],
    assets: File[],
    pattern: string // e.g., "workcard_{day}.html"
  ): Map<string, File> {
    
    const mappings = new Map<string, File>();
    
    // Convert pattern to regex
    const regex = new RegExp(
      pattern.replace(/\{(\w+)\}/g, '(?<$1>\\w+)')
    );
    
    for (const asset of assets) {
      const match = asset.name.match(regex);
      if (!match || !match.groups) continue;
      
      // Find placeholder with matching groups
      const placeholder = placeholders.find(p => {
        const pGroups = this.parsePlaceholder(p.placeholder);
        return Object.values(match.groups!).every(v => 
          pGroups.includes(v.toLowerCase())
        );
      });
      
      if (placeholder) {
        mappings.set(placeholder.placeholder, asset);
      }
    }
    
    return mappings;
  }
  
  private static parsePlaceholder(placeholder: string): string[] {
    // {{WORKCARD_DAY_01}} → ["workcard", "day", "01"]
    return placeholder
      .replace(/[{}]/g, '')
      .toLowerCase()
      .split(/[_\s-]+/)
      .filter(Boolean);
  }
}

// exportEngine.ts (extended)

export class ExportEngine {
  
  static async exportWithProfile(
    manifest: AssemblyManifest,
    assembledContent: string,
    assets: ProcessedAsset[],
    profileId: string
  ): Promise<ExportResult> {
    
    const profile = EXPORT_PROFILES.find(p => p.id === profileId);
    if (!profile) throw new Error('Profile not found');
    
    // Apply profile transformations
    const transformed = await this.applyProfileTransforms(
      assembledContent,
      assets,
      profile
    );
    
    // Export based on format
    switch (profile.format) {
      case 'html':
        return this.exportHTML(manifest, transformed, assets);
      case 'epub':
        return this.exportEPUB(manifest, transformed, assets, profile.config);
      case 'pdf':
        return this.exportPDF(manifest, transformed, profile.config);
    }
  }
  
  private static async applyProfileTransforms(
    content: string,
    assets: ProcessedAsset[],
    profile: ExportProfile
  ): Promise<string> {
    
    let transformed = content;
    
    // Font size adjustment
    if (profile.config.fontSize) {
      transformed = transformed.replace(
        /<body/,
        `<body style="font-size: ${profile.config.fontSize}px; line-height: ${profile.config.lineHeight}"`
      );
    }
    
    // Color mode conversion
    if (profile.config.colorMode === 'grayscale') {
      transformed = transformed.replace(
        /<style>/,
        '<style>* { filter: grayscale(100%); }'
      );
    }
    
    // Image optimization
    if (profile.config.optimizeImages) {
      for (const asset of assets.filter(a => a.type === 'raster')) {
        const optimized = await this.optimizeImage(asset, profile);
        transformed = transformed.replaceAll(asset.processedURL, optimized.processedURL);
      }
    }
    
    // Accessibility enhancements
    if (profile.config.accessibility) {
      transformed = this.addAccessibilityAttributes(transformed);
    }
    
    return transformed;
  }
  
  private static addAccessibilityAttributes(html: string): string {
    // Add ARIA labels, semantic HTML, etc.
    return html
      .replace(/<img/g, '<img role="img"')
      .replace(/<button/g, '<button role="button"')
      .replace(/<input/g, '<input aria-required="false"');
  }
}

static async listSnapshots(manifestTitle: string): Promise<ProjectSnapshot[]> {
const keys = await window.storage.list(this.STORAGE_PREFIX);
if (!keys) return [];

const snapshots = await Promise.all(
  keys.keys.map(key => this.loadSnapshot(key.replace(this.STORAGE_PREFIX, '')))

  .filter((s): s is ProjectSnapshot => s !== null)
  .filter(s => s.manifest.metadata.title === manifestTitle)
  .sort((a, b) => new Date(b.timestamp).getTime() - new Date(a.timestamp).getTime());

```typescript
      if (!manifestResult || !mappingsResult) return null;

      };

      return null;
```

```typescript
// services/monitoring/performanceMonitor.ts

export class PerformanceMonitor {
  
  private static metrics: Map<string, number[]> = new Map();
  
  static measure<T>(operation: string, fn: () => T | Promise<T>): T | Promise<T> {
    const start = performance.now();
    
    const result = fn();
    
    if (result instanceof Promise) {
      return result.finally(() => {
        this.recordMetric(operation, performance.now() - start);
      }) as Promise<T>;
    } else {
      this.recordMetric(operation, performance.now() - start);
      return result;
    }
  }
  
  private static recordMetric(operation: string, duration: number) {
    if (!this.metrics.has(operation)) {
      this.metrics.set(operation, []);
    }
    this.metrics.get(operation)!.push(duration);
    
    // Keep last 100 measurements
    if (this.metrics.get(operation)!.length > 100) {
      this.metrics.get(operation)!.shift();
    }
  }
  
  static getStats(operation: string) {
    const measurements = this.metrics.get(operation) || [];
    if (measurements.length === 0) return null;
    
    const sorted = [...measurements].sort((a, b) => a - b);
    return {
      min: sorted[0],
      max: sorted[sorted.length - 1],
      avg: measurements.reduce((a, b) => a + b) / measurements.length,
      p50: sorted[Math.floor(sorted.length * 0.5)],
      p95: sorted[Math.floor(sorted.length * 0.95)],
      count: measurements.length
    };
  }
  
  static getAllStats() {
    return Array.from(this.metrics.keys()).map(op => ({
      operation: op,
      ...this.getStats(op)
    }));
  }
}
```

```
## 5. SECURITY & SANDBOXING

### 5.1 XSS Prevention Strategy

**Threat model:**
User wstrzykuje złośliwy HTML → XSS → kradzież danych / session hijacking.

**Defense layers:**
```
USER HTML INPUT
    ↓
[1] FORBIDDEN PATTERN CHECK
    - Regex dla: javascript:, on*, <script>, eval()
    - Reject natychmiast
    ↓
[2] DOMPURIFY SANITIZATION
    - Whitelist allowed tags/attrs
    - Remove wszystkie event handlers
    - Encode dangerous chars
    ↓
[3] SANDBOX WRAPPER
    - CSS isolation: isolation: isolate
    - Contain: layout style paint
    - No script execution context
    ↓
[4] CONTENT SECURITY POLICY (CSP)
    - script-src 'self'
    - object-src 'none'
    - frame-ancestors 'none'
    ↓
SAFE RENDERING
5.2 Allowed vs Forbidden
Allowed tags:

typescript
['div', 'span', 'p', 'h1-h6', 'ul', 'ol', 'li', 
 'table', 'tr', 'td', 'th', 'input', 'textarea', 
 'label', 'button', 'select', 'img', 'audio', 'video']
Forbidden tags:

typescript
['script', 'iframe', 'object', 'embed', 'applet', 
 'link', 'style', 'meta', 'base']
Forbidden attributes:

typescript
['onclick', 'onerror', 'onload', 'onmouseover',

Dla max security, wrap w Shadow DOM:

    <style>

CSS nie leakuje na zewnątrz
External CSS nie wpływa na workcard
JS nie może dostać się do parent DOM

## 7. PERFORMANCE GUIDELINES

### 7.1 Performance Targets

| Operation | Target | Method |
|-----------|--------|--------|
| Parse manifest | <100ms | Sync JSON.parse |
| Validate schema | <200ms | AJV validator |
| Process 1 image | <500ms | OffscreenCanvas |
| Process 1 HTML | <300ms | DOMPurify |
| Map 50 assets | <2s | Async parallel |
| Validate (local) | <500ms | Sync checks |
| Validate (AI) | <5s | Single API call |
| Inject placeholders | <3s | String replacement batch |
| Export HTML | <1s | Template + bundle |
| Export EPUB | <5s | Worker thread |
| Export PDF | <15s | Worker thread |

### 7.2 Optimization Strategies

**1. Lazy Loading Assets**
```typescript
// Don't load all assets upfront
const asset = mappings.get(placeholder);
if (!asset.loaded) {
  await loadAsset(asset); // Load on-demand
}
```

**2. Batch Processing**
```typescript
// Process images in parallel
await Promise.all(
  images.map(img => AssetProcessor.process(img, 'raster', config))
);
```

**3. Worker Threads dla Heavy Operations**
```typescript
// PDF generation off main thread
const worker = new Worker('/workers/pdf.worker.js');
worker.postMessage({ content, config });
```

**4. Incremental Rendering**
```typescript
// Don't wait for all sections, render progressively
sections.forEach((section, i) => {
  setTimeout(() => renderSection(section), i * 100);
});
```

**5. Memoization**
```typescript
const processedCache = new Map<string, ProcessedAsset>();

async function getProcessed(file: File): Promise<ProcessedAsset> {
  const key = `${file.name}_${file.size}`;
  if (processedCache.has(key)) {
    return processedCache.get(key)!;
  }
  const processed = await AssetProcessor.process(file);
  processedCache.set(key, processed);
  return processed;
}
```

### 7.3 Memory Management

**Problem:** 50+ high-res images = memory leak.

**Solution:**
```typescript
// Revoke object URLs after use
function cleanup(assets: ProcessedAsset[]) {
  assets.forEach(asset => {
    if (asset.processedURL.startsWith('blob:')) {
      URL.revokeObjectURL(asset.processedURL);
    }
  });
}

// Call after export
await ExportEngine.exportPDF(...);
cleanup(processedAssets);
```

## 10. TROUBLESHOOTING

### 10.1 Common Issues

**Problem: Manifest validation fails**
Error: Invalid manifest: data.sections[0].chapter should be >= 1

**Solution:** Check chapter numbering starts at 1, is sequential.

---

**Problem: Asset not mapping**
Warning: Placeholder {{WORKCARD_01}} not resolved

**Solution:** 
1. Check filename matches manifest path exactly
2. Use batch auto-map
3. Verify file uploaded successfully

---

**Problem: HTML injection fails / blank output**
Error: DOMPurify removed all content

**Solution:**
1. Check HTML doesn't contain `<script>` tags
2. Remove inline event handlers (`onclick=...`)
3. Use allowed tags only

---

**Problem: PDF export hangs**
Worker timeout after 30s

**Solution:**
1. Reduce image count/size
2. Split into multiple PDFs
3. Use "draft" quality profile

---

**Problem: Memory leak on large projects**
Out of memory error after processing 100+ images

**Solution:**
```typescript
// Revoke blob URLs periodically
setInterval(() => {
  processedAssets.forEach(a => {
    if (!a.inUse) URL.revokeObjectURL(a.processedURL);
  });
}, 60000); // Every minute
```
