                  <label className="block text-sm font-semibold text-gray-700 mb-2">Tytuł rozdziału</label>
                  <input
                    type="text"
                    value={chapter.title}
                    onChange={(e) => updateChapter(chapter.id, 'title', e.target.value)}
                    className="w-full px-4 py-3 border-2 border-gray-200 rounded-lg focus:border-indigo-500 focus:outline-none transition-colors"
                    placeholder="Tytuł rozdziału..."
                  />
                </div>
                
                <div>
                  <label className="block text-sm font-semibold text-gray-700 mb-2">Treść</label>
                  <textarea
                    value={chapter.content}
                    onChange={(e) => updateChapter(chapter.id, 'content', e.target.value)}
                    className="w-full px-4 py-3 border-2 border-gray-200 rounded-lg focus:border-indigo-500 focus:outline-none h-48 resize-y"
                    placeholder="Wklej swoją treść tutaj... (rozdziel akapity podwójnym enterem)"
                  />
                </div>

                <div>
                  <label className="block text-sm font-semibold text-gray-700 mb-2">Obrazy do rozdziału</label>
                  <input
                    type="file"
                    id={`img-${chapter.id}`}
                    onChange={(e) => handleImageUpload(chapter.id, e)}
                    multiple
                    accept="image/*"
                    className="hidden"
                  />
                  <label
                    htmlFor={`img-${chapter.id}`}
                    className="flex items-center justify-center gap-2 px-4 py-3 bg-indigo-50 text-indigo-600 rounded-lg hover:bg-indigo-100 transition-colors cursor-pointer"
                  >
                    <ImagePlus className="w-5 h-5" />
                    Dodaj obrazy
                  </label>
                  
                  {chapter.images.length > 0 && (
                    <div className="grid grid-cols-2 md:grid-cols-3 gap-4 mt-4">
                      {chapter.images.map((img, imgIdx) => (
                        <div key={imgIdx} className="relative group">
                          <img src={img} alt={`Obraz ${imgIdx + 1}`} className="w-full h-32 object-cover rounded-lg shadow-md" />
                          <button
                            onClick={() => removeImage(chapter.id, imgIdx)}
                            className="absolute top-2 right-2 bg-red-500 text-white rounded-full w-7 h-7 flex items-center justify-center opacity-0 group-hover:opacity-100 transition-opacity shadow-lg"
                          >
                            ×
                          </button>
                        </div>
                      ))}
                    </div>
                  )}
                </div>
              </div>
            </div>
          ))}
        </div>

        <div className="flex flex-wrap gap-3 md:gap-4 mb-8">
          <button
            onClick={addChapter}
            className="flex items-center gap-2 px-5 py-3 bg-indigo-600 text-white rounded-lg hover:bg-indigo-700 transition-colors shadow-lg text-sm md:text-base"
          >
            <Type className="w-5 h-5" />
            Dodaj rozdział
          </button>
          
          <button
            onClick={() => setPreview(!preview)}
            disabled={chapters.length === 0}
            className="flex items-center gap-2 px-5 py-3 bg-green-600 text-white rounded-lg hover:bg-green-700 transition-colors shadow-lg disabled:opacity-50 disabled:cursor-not-allowed text-sm md:text-base"
          >
            <Eye className="w-5 h-5" />
            {preview ? 'Ukryj podgląd' : 'Podgląd'}
          </button>
          
          <button
            onClick={generatePDF}
            disabled={!bookTitle || chapters.length === 0}
            className="flex items-center gap-2 px-5 py-3 bg-purple-600 text-white rounded-lg hover:bg-purple-700 transition-colors shadow-lg disabled:opacity-50 disabled:cursor-not-allowed text-sm md:text-base"
          >
            <Download className="w-5 h-5" />
            Eksportuj do PDF
          </button>
        </div>

        {preview && chapters.length > 0 && (
          <div className="bg-white rounded-2xl shadow-xl p-6 md:p-8">
            <h3 className="text-xl md:text-2xl font-bold text-gray-800 mb-6">Podgląd ebooka</h3>
            <div 
              className="p-8 md:p-12 rounded-xl overflow-hidden"
              style={{ 
                background: currentTheme.bg,
                fontFamily: currentTheme.font,
                color: currentTheme.primary
              }}
            >
              <div className="text-center mb-16">
                <h1 
                  className="text-3xl md:text-5xl mb-4 font-bold"
                  style={{ 
                    fontFamily: currentTheme.titleFont,
                    color: currentTheme.primary 
                  }}
                >
                  {bookTitle}
                </h1>
                <div 
                  className="w-32 md:w-48 h-1 mx-auto my-8"
                  style={{ background: currentTheme.accent }}
                />
                <p className="text-xl md:text-2xl italic" style={{ color: currentTheme.secondary }}>
                  autor: {author}
                </p>
              </div>

              {chapters.map((ch) => (
                <div key={ch.id} className="mb-16">
                  <h2 
                    className="text-2xl md:text-3xl mb-6 pb-2 border-b-2 font-bold"
                    style={{ 
                      fontFamily: currentTheme.titleFont,
                      color: currentTheme.accent,
                      borderColor: currentTheme.accent
                    }}
                  >
                    {ch.title}
                  </h2>
                  <div className="text-base md:text-lg leading-relaxed">
                    {ch.content.split('\n\n').map((para, pIdx) => (
                      <p key={pIdx} className="mb-4 text-justify first:indent-0 indent-8">
                        {para}
                      </p>
                    ))}
                    {ch.images.map((img, imgIdx) => (
                      <div key={imgIdx} className="my-8">
                        <img 
                          src={img} 
                          alt={`Ilustracja ${imgIdx + 1}`} 
                          className="max-w-full h-auto mx-auto rounded-lg shadow-xl"
                        />
                        <p className="text-center italic mt-2 text-sm" style={{ color: currentTheme.secondary }}>
                          Ilustracja {imgIdx + 1}
                        </p>
                      </div>
                    ))}
                  </div>
                </div>
              ))}
            </div>
          </div>
        )}
      </div>
    </div>
  );
};

export default EbookGenerator;
