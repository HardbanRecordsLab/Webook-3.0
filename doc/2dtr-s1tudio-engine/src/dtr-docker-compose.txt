version: '3.8'

services:
  # PostgreSQL Database
  postgres:
    image: postgres:15-alpine
    container_name: dtr-postgres
    environment:
      POSTGRES_USER: dtr_user
      POSTGRES_PASSWORD: dtr_password_2024
      POSTGRES_DB: dtr_studio
    volumes:
      - postgres_data:/var/lib/postgresql/data
    ports:
      - "5432:5432"
    networks:
      - dtr-network
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U dtr_user"]
      interval: 10s
      timeout: 5s
      retries: 5

  # Backend API (Node.js + Express)
  backend:
    build:
      context: ./backend
      dockerfile: Dockerfile
    container_name: dtr-backend
    environment:
      NODE_ENV: production
      PORT: 5000
      DATABASE_URL: postgresql://dtr_user:dtr_password_2024@postgres:5432/dtr_studio
      JWT_SECRET: dtr_jwt_secret_key_2024_change_in_production
      UPLOAD_DIR: /app/uploads
    volumes:
      - uploads:/app/uploads
      - ./backend:/app
      - /app/node_modules
    ports:
      - "5000:5000"
    depends_on:
      postgres:
        condition: service_healthy
    networks:
      - dtr-network
    command: npm run dev

  # Frontend React App
  frontend:
    build:
      context: ./frontend
      dockerfile: Dockerfile
    container_name: dtr-frontend
    environment:
      REACT_APP_API_URL: http://localhost:5000
    volumes:
      - ./frontend:/app
      - /app/node_modules
    ports:
      - "3000:3000"
    depends_on:
      - backend
    networks:
      - dtr-network
    command: npm start

volumes:
  postgres_data:
    driver: local
  uploads:
    driver: local

networks:
  dtr-network:
    driver: bridge
