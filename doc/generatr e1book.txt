3. MANIFEST SERVICE (manifestService.ts)
typescript
import Ajv from 'ajv';
import manifestSchema from '../schemas/manifest.schema.json';

const ajv = new Ajv();
const validateSchema = ajv.compile(manifestSchema);

export class ManifestService {
  
  // Parse i walidacja manifestu
  static async parseManifest(file: File): Promise<AssemblyManifest> {
    const text = await file.text();
    const data = JSON.parse(text);
    
    if (!validateSchema(data)) {
      throw new Error(`Invalid manifest: ${JSON.stringify(validateSchema.errors)}`);
    }
    
    return data as AssemblyManifest;
  }
  
  // Ekstrakcja wszystkich placeholderów
  static extractPlaceholders(manifest: AssemblyManifest): PlaceholderMapping[] {
    const mappings: PlaceholderMapping[] = [];
    
    manifest.sections.forEach(section => {
      Object.entries(section.placeholders).forEach(([key, path]) => {
        mappings.push({
          placeholder: key,
          type: this.detectAssetType(path),
          sourcePath: path,
          resolved: false
        });
      });
    });
    
    return mappings;
  }
  
  // Detekcja typu assetu z rozszerzenia
  private static detectAssetType(path: string): AssetType {
    const ext = path.split('.').pop()?.toLowerCase();
    
    if (['html', 'htm'].includes(ext!)) return 'workcard-html';
    if (['mp3', 'wav', 'ogg'].includes(ext!)) return 'audio';
    if (['jpg', 'jpeg', 'png', 'webp'].includes(ext!)) return 'raster';
    if (['svg'].includes(ext!)) return 'vector';
    
    return 'source';
  }
  
  // Walidacja struktury (bez AI)
  static validateStructure(manifest: AssemblyManifest): ValidationError[] {
    const errors: ValidationError[] = [];
    
    // Check sections order
    const chapters = manifest.sections.map(s => s.chapter);
    if (chapters.some((ch, i) => i > 0 && ch <= chapters[i-1])) {
      errors.push({
        type: 'INVALID_PATH',
        placeholder: 'sections',
        message: 'Chapters must be in ascending order',
        severity: 'ERROR'
      });
    }
    
    // Check duplicate placeholders
    const allPlaceholders = manifest.sections.flatMap(s => 
      Object.keys(s.placeholders)
    );
    const duplicates = allPlaceholders.filter((p, i) => 
      allPlaceholders.indexOf(p) !== i
    );
    
    if (duplicates.length > 0) {
      errors.push({
        type: 'INVALID_PATH',
        placeholder: duplicates[0],
        message: `Duplicate placeholder: ${duplicates[0]}`,
        severity: 'ERROR'
      });
    }
    
    return errors;
  }
}
4. PLACEHOLDER PARSER (placeholderParser.ts)
typescript
export class PlaceholderParser {
  
  // Wykryj wszystkie {{PLACEHOLDERS}} w tekście
  static detectInText(text: string): string[] {
    const regex = /\{\{([A-Z0-9_]+)\}\}/g;
    const matches = [...text.matchAll(regex)];
    return [...new Set(matches.map(m => `{{${m[1]}}}`))];
  }
  
  // Zamień placeholdery na rzeczywisty content
  static async injectContent(
    text: string,
    mappings: Map<string, ProcessedAsset>
  ): Promise<string> {
    let result = text;
    
    for (const [placeholder, asset] of mappings) {
      const replacement = await this.generateReplacement(asset);
      result = result.replaceAll(placeholder, replacement);
    }
    
    return result;
  }
  
  // Generuj HTML replacement w zależności od typu
  private static async generateReplacement(asset: ProcessedAsset): Promise<string> {
    switch (asset.type) {
      case 'workcard-html':
        return `
          <section 
            data-workcard-id="${asset.id}"
            data-checksum="${asset.checksum}"
            class="workcard-inject"
            style="isolation: isolate;"
          >
            ${await fetch(asset.processedURL).then(r => r.text())}
          </section>
        `;
      
      case 'audio':
        return `
          <audio 
            controls 
            data-asset-id="${asset.id}"
            class="audio-inject"
          >
            <source src="${asset.processedURL}" type="audio/mpeg">
          </audio>
        `;
      
      case 'raster':
        return `
          <figure data-asset-id="${asset.id}" class="image-inject">
            <img 
              src="${asset.processedURL}" 
              alt="Asset ${asset.id}"
              loading="lazy"
            />
          </figure>
        `;
      
      default:
        return `<!-- UNSUPPORTED ASSET TYPE: ${asset.type} -->`;
    }
  }
}

export class AssemblyValidator {
  
  // AI-powered semantic validation (TYLKO validation, zero content generation)
  static async validateWithAI(
    manifest: AssemblyManifest,
    mappings: PlaceholderMapping[]
  ): Promise<ValidationResult> {
    
    const prompt = `
STRICT VALIDATION TASK - DO NOT GENERATE CONTENT

Manifest sections: ${manifest.sections.length}
Expected placeholders: ${mappings.length}

Placeholders status:
${mappings.map(m => `${m.placeholder}: ${m.resolved ? 'RESOLVED' : 'MISSING'}`).join('\n')}

YOUR TASK:
1. Identify missing assets
2. Detect orphaned files
3. Check naming consistency
4. Report errors

RETURN ONLY JSON:
{
  "status": "COMPLETE" | "INCOMPLETE",
  "missing": ["{{PLACEHOLDER}}", ...],
  "orphaned": ["filename.ext", ...],
  "warnings": ["message", ...]
}

DO NOT:
- Generate placeholder content
- Suggest layouts
- Modify manifest
- Create assets
`;

    const response = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1000,
        messages: [{ role: 'user', content: prompt }]
      })
    });
    
    const data = await response.json();
    const result = JSON.parse(
      data.content[0].text.replace(/```json|```/g, '').trim()
    );
    
    return {
      status: result.status,
      total: mappings.length,
      mapped: mappings.filter(m => m.resolved).length,
      missing: mappings.filter(m => !m.resolved),
      orphaned: result.orphaned || [],
      errors: result.warnings?.map((w: string) => ({
        type: 'MISSING_ASSET',
        placeholder: '',
        message: w,
        severity: 'WARNING'
      })) || []
    };
  }
  
  // Lokalna walidacja (bez AI)
  static validateLocal(mappings: PlaceholderMapping[]): ValidationResult {
    return {
      status: mappings.every(m => m.resolved) ? 'COMPLETE' : 'INCOMPLETE',
      total: mappings.length,
      mapped: mappings.filter(m => m.resolved).length,
      missing: mappings.filter(m => !m.resolved),
      orphaned: [],
      errors: []
    };
  }
}
6. EXPORT ENGINE (exportEngine.ts)

export class ExportEngine {
  
  // Export do HTML (single file)
  static async exportHTML(
    manifest: AssemblyManifest,
    assembledContent: string,
    assets: ProcessedAsset[]
  ): Promise<ExportResult> {
    
    const html = `
<!DOCTYPE html>
<html lang="${manifest.metadata.language}">
<head>
  <meta charset="UTF-8">
  <meta name="author" content="${manifest.metadata.author}">
  <title>${manifest.metadata.title}</title>
  <style>${await this.bundleCSS(manifest.assets.css || [])}</style>
</head>
<body>
  <main class="ebook-container">
    ${assembledContent}
  </main>
  <script>${await this.bundleScripts(manifest.assets.scripts || [])}</script>
</body>
</html>
    `;
    
    const blob = new Blob([html], { type: 'text/html' });
    const url = URL.createObjectURL(blob);
    
    return {
      success: true,
      format: 'html',
      size: blob.size,
      downloadURL: url,
      metadata: {
        assets: assets.length,
        generatedAt: new Date().toISOString()
      }
    };
  }
  
  // Export do EPUB (via worker)
  static async exportEPUB(
    manifest: AssemblyManifest,
    assembledContent: string,
    assets: ProcessedAsset[]
  ): Promise<ExportResult> {
    
    const worker = new Worker(new URL('../workers/epub.worker.ts', import.meta.url));
    
    return new Promise((resolve, reject) => {
      worker.postMessage({
        manifest,
        content: assembledContent,
        assets
      });
      
      worker.onmessage = (e) => {
        if (e.data.success) {
          resolve(e.data);
        } else {
          reject(e.data.error);
        }
        worker.terminate();
      };
    });
  }
  
  // Export do PDF (via worker - heavy processing)
  static async exportPDF(
    manifest: AssemblyManifest,
    assembledContent: string,
    config: ExportConfig
  ): Promise<ExportResult> {
    
    const worker = new Worker(new URL('../workers/pdf.worker.ts', import.meta.url));
    
    return new Promise((resolve, reject) => {
      worker.postMessage({
        manifest,
        content: assembledContent,
        config
      });
      
      worker.onmessage = (e) => {
        if (e.data.success) {
          resolve(e.data);
        } else {
          reject(e.data.error);
        }
        worker.terminate();
      };
    });
  }
  
  private static async bundleCSS(cssPaths: string[]): Promise<string> {
    const styles = await Promise.all(
      cssPaths.map(path => fetch(path).then(r => r.text()))
    );
    return styles.join('\n\n');
  }
  
  private static async bundleScripts(scriptPaths: string[]): Promise<string> {
    const scripts = await Promise.all(
      scriptPaths.map(path => fetch(path).then(r => r.text()))
    );
    return scripts.join('\n\n');
  }
}

// services/assembly/htmlSanitizer.ts

import DOMPurify from 'dompurify';

export class HTMLSanitizer {
  
  private static ALLOWED_TAGS = [
    'div', 'span', 'p', 'h1', 'h2', 'h3', 'h4', 'h5', 'h6',
    'ul', 'ol', 'li', 'table', 'tr', 'td', 'th',
    'input', 'textarea', 'label', 'button', 'select', 'option',
    'img', 'audio', 'video', 'canvas'
  ];
  
  private static ALLOWED_ATTRS = [
    'class', 'id', 'style', 'data-*', 
    'type', 'value', 'placeholder', 'name',
    'src', 'alt', 'width', 'height'
  ];
  
  private static FORBIDDEN_PATTERNS = [
    /javascript:/gi,
    /on\w+=/gi,  // onclick, onerror, etc.
    /<script/gi,
    /<iframe/gi,
    /eval\(/gi
  ];
  
  // Sanitize HTML przed injection
  static sanitize(html: string, strict: boolean = true): string {
    
    // 1. Check for forbidden patterns
    for (const pattern of this.FORBIDDEN_PATTERNS) {
      if (pattern.test(html)) {
        throw new AssemblyError(
          'SECURITY_VIOLATION',
          'ERROR',
          `Forbidden pattern detected: ${pattern.source}`
        );
      }
    }
    
    // 2. DOMPurify clean
    const clean = DOMPurify.sanitize(html, {
      ALLOWED_TAGS: strict ? this.ALLOWED_TAGS : undefined,
      ALLOWED_ATTR: this.ALLOWED_ATTRS,
      ALLOW_DATA_ATTR: true,
      FORBID_TAGS: ['script', 'iframe', 'object', 'embed'],
      FORBID_ATTR: ['onerror', 'onload', 'onclick']
    });
    
    // 3. Wrap w sandboxed container
    return `
      <div 
        class="workcard-sandbox" 
        data-sanitized="true"
        style="
          all: initial;
          display: block;
          isolation: isolate;
          contain: layout style paint;
        "
      >
        ${clean}
      </div>
    `;
  }
  
  // CSS isolation via Shadow DOM (optional)
  static wrapInShadowDOM(html: string, css: string): string {
    return `
      <div class="shadow-host">
        <template shadowrootmode="open">
          <style>
            :host { all: initial; display: block; }
            ${css}
          </style>
          ${html}
        </template>
      </div>
    `;
  }
}

// services/assembly/assetProcessor.ts

export class AssetProcessor {
  
  // Process uploaded file → ProcessedAsset
  static async process(
    file: File,
    type: AssetType,
    config: ProcessingConfig
  ): Promise<ProcessedAsset> {
    
    switch (type) {
      case 'workcard-html':
        return this.processHTML(file, config);
      case 'raster':
        return this.processImage(file, config);
      case 'audio':
        return this.processAudio(file, config);
      default:
        return this.processGeneric(file);
    }
  }
  
  // HTML processing
  private static async processHTML(
    file: File, 
    config: ProcessingConfig
  ): Promise<ProcessedAsset> {
    
    const html = await file.text();
    
    // Extract inline CSS
    const cssMatch = html.match(/<style[^>]*>([\s\S]*?)<\/style>/gi);
    const inlineCSS = cssMatch ? cssMatch.join('\n') : '';
    
    // Remove <html>, <head>, <body> wrappers
    const bodyContent = html
      .replace(/<\/?html[^>]*>/gi, '')
      .replace(/<head[^>]*>[\s\S]*?<\/head>/gi, '')
      .replace(/<\/?body[^>]*>/gi, '');
    
    // Sanitize
    const sanitized = HTMLSanitizer.sanitize(bodyContent);
    
    // Generate blob
    const blob = new Blob([sanitized], { type: 'text/html' });
    const url = URL.createObjectURL(blob);
    
    return {
      id: this.generateAssetId(),
      originalPath: file.name,
      processedURL: url,
      type: 'workcard-html',
      size: blob.size,
      checksum: await this.generateChecksum(blob),
      metadata: {
        inlineCSS,
        sanitized: true
      }
    };
  }
  
  // Image optimization
  private static async processImage(
    file: File,
    config: ProcessingConfig
  ): Promise<ProcessedAsset> {
    
    const img = await createImageBitmap(file);
    
    const canvas = new OffscreenCanvas(
      config.maxWidth || img.width,
      config.maxHeight || img.height
    );
    
    const ctx = canvas.getContext('2d')!;
    
    // Resize preserving aspect ratio
    const ratio = Math.min(
      canvas.width / img.width,
      canvas.height / img.height
    );
    
    const w = img.width * ratio;
    const h = img.height * ratio;
    
    ctx.drawImage(img, 0, 0, w, h);
    
    // Convert to blob
    const blob = await canvas.convertToBlob({
      type: config.format || 'image/webp',
      quality: config.quality || 0.85
    });
    
    const url = URL.createObjectURL(blob);
    
    return {
      id: this.generateAssetId(),
      originalPath: file.name,
      processedURL: url,
      type: 'raster',
      size: blob.size,
      checksum: await this.generateChecksum(blob),
      metadata: {
        originalSize: file.size,
        compressionRatio: ((1 - blob.size / file.size) * 100).toFixed(1) + '%',
        dimensions: { width: w, height: h }
      }
    };
  }
  
  // Audio normalization
  private static async processAudio(
    file: File,
    config: ProcessingConfig
  ): Promise<ProcessedAsset> {
    
    // Dla audio możemy tylko validate + convert format
    const audioContext = new AudioContext();
    const arrayBuffer = await file.arrayBuffer();
    const audioBuffer = await audioContext.decodeAudioData(arrayBuffer);
    
    // Optional: normalize volume, trim silence, etc.
    // For now just pass-through
    
    const url = URL.createObjectURL(file);
    
    return {
      id: this.generateAssetId(),
      originalPath: file.name,
      processedURL: url,
      type: 'audio',
      size: file.size,
      checksum: await this.generateChecksum(file),
      metadata: {
        duration: audioBuffer.duration,
        sampleRate: audioBuffer.sampleRate,
        channels: audioBuffer.numberOfChannels
      }
    };
  }
  
  // Checksum generation (integrity verification)
  private static async generateChecksum(blob: Blob): Promise<string> {
    const buffer = await blob.arrayBuffer();
    const hashBuffer = await crypto.subtle.digest('SHA-256', buffer);
    const hashArray = Array.from(new Uint8Array(hashBuffer));
    return hashArray.map(b => b.toString(16).padStart(2, '0')).join('');
  }
  
  private static generateAssetId(): string {
    return `asset_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
  }
}

interface ProcessingConfig {
  maxWidth?: number;
  maxHeight?: number;
  quality?: number;
  format?: 'image/webp' | 'image/jpeg' | 'image/png';
}

// services/assembly/incrementalValidator.ts

export class IncrementalValidator {
  
  private mappings = new Map<string, PlaceholderMapping>();
  private listeners: Array<(result: PartialValidationResult) => void> = [];
  
  // Subscribe to validation updates
  onUpdate(callback: (result: PartialValidationResult) => void) {
    this.listeners.push(callback);
  }
  
  // Add/update single mapping
  async updateMapping(placeholder: string, asset: ProcessedAsset | null) {
    
    const mapping = this.mappings.get(placeholder);
    if (!mapping) return;
    
    mapping.asset = asset;
    mapping.resolved = asset !== null;
    
    // Instant validation
    const validation = this.validateSingle(mapping);
    
    // Notify listeners
    this.notifyUpdate({
      placeholder,
      status: validation.status,
      errors: validation.errors,
      progress: this.calculateProgress()
    });
  }
  
  // Validate single mapping
  private validateSingle(mapping: PlaceholderMapping): ValidationResult {
    const errors: ValidationError[] = [];
    
    if (!mapping.resolved) {
      errors.push({
        type: 'MISSING_ASSET',
        placeholder: mapping.placeholder,
        message: `Asset not mapped for ${mapping.placeholder}`,
        severity: 'ERROR'
      });
    }
    
    if (mapping.asset) {
      // Type mismatch check
      const expectedType = this.inferExpectedType(mapping.placeholder);
      if (expectedType && mapping.asset.type !== expectedType) {
        errors.push({
          type: 'UNSUPPORTED_TYPE',
          placeholder: mapping.placeholder,
          message: `Expected ${expectedType}, got ${mapping.asset.type}`,
          severity: 'WARNING'
        });
      }
      
      // Size check
      if (mapping.asset.size > 10 * 1024 * 1024) { // 10MB
        errors.push({
          type: 'INVALID_PATH',
          placeholder: mapping.placeholder,
          message: `Asset size ${(mapping.asset.size / 1024 / 1024).toFixed(1)}MB exceeds 10MB limit`,
          severity: 'WARNING'
        });
      }
    }
    
    return {
      status: errors.some(e => e.severity === 'ERROR') ? 'INCOMPLETE' : 'COMPLETE',
      total: 1,
      mapped: mapping.resolved ? 1 : 0,
      missing: mapping.resolved ? [] : [mapping],
      orphaned: [],
      errors
    };
  }
  
  // Infer expected type from placeholder name
  private inferExpectedType(placeholder: string): AssetType | null {
    if (/WORKCARD|CARD|WORKSHEET/i.test(placeholder)) return 'workcard-html';
    if (/AUDIO|SOUND|MUSIC/i.test(placeholder)) return 'audio';
    if (/IMAGE|IMG|PHOTO|PICTURE/i.test(placeholder)) return 'raster';
    return null;
  }
  
  // Calculate overall progress
  private calculateProgress(): number {
    const total = this.mappings.size;
    const resolved = Array.from(this.mappings.values()).filter(m => m.resolved).length;
    return total > 0 ? (resolved / total) * 100 : 0;
  }
  
  private notifyUpdate(result: PartialValidationResult) {
    this.listeners.forEach(cb => cb(result));
  }
}

interface PartialValidationResult {
  placeholder: string;
  status: 'COMPLETE' | 'INCOMPLETE';
  errors: ValidationError[];
  progress: number;
}

// services/assembly/commandStack.ts

interface Command {
  execute(): Promise<void>;
  undo(): Promise<void>;
  description: string;
}

export class CommandStack {
  
  private stack: Command[] = [];
  private pointer: number = -1;
  
  async execute(command: Command) {
    // Remove future commands if we're in the middle of history
    this.stack = this.stack.slice(0, this.pointer + 1);
    
    await command.execute();
    
    this.stack.push(command);
    this.pointer++;
  }
  
  async undo() {
    if (!this.canUndo()) return;
    
    await this.stack[this.pointer].undo();
    this.pointer--;
  }
  
  async redo() {
    if (!this.canRedo()) return;
    
    this.pointer++;
    await this.stack[this.pointer].execute();
  }
  
  canUndo(): boolean {
    return this.pointer >= 0;
  }
  
  canRedo(): boolean {
    return this.pointer < this.stack.length - 1;
  }
  
  getHistory(): string[] {
    return this.stack.map(c => c.description);
  }
}

// Example: MapAssetCommand
export class MapAssetCommand implements Command {

  }

// services/assembly/batchMapper.ts

export class BatchMapper {
  
  // Auto-map by naming convention
  static autoMapByNaming(
    placeholders: PlaceholderMapping[],
    assets: File[]
  ): Map<string, File> {
    
    const mappings = new Map<string, File>();
    
    for (const placeholder of placeholders) {
      // Extract semantic parts
      // {{WORKCARD_DAY_01}} → "workcard", "day", "01"
      const parts = this.parsePlaceholder(placeholder.placeholder);
      
      // Find best matching file
      const match = assets.find(file => {
        const fileName = file.name.toLowerCase();
        return parts.every(part => fileName.includes(part));
      });
      
      if (match) {
        mappings.set(placeholder.placeholder, match);
      }
    }
    
    return mappings;
  }
  
  // Bulk upload folder
  static async uploadFolder(
    folderInput: HTMLInputElement
  ): Promise<Map<string, File>> {
    
    const files = Array.from(folderInput.files || []);
    const organized = new Map<string, File>();
    
    files.forEach(file => {
      // Preserve folder structure
      const path = file.webkitRelativePath || file.name;
      organized.set(path, file);
    });
    
    return organized;
  }
  
  // Pattern-based mapping
  static mapByPattern(
    placeholders: PlaceholderMapping[],
    assets: File[],
    pattern: string // e.g., "workcard_{day}.html"
  ): Map<string, File> {
    
    const mappings = new Map<string, File>();
    
    // Convert pattern to regex
    const regex = new RegExp(
      pattern.replace(/\{(\w+)\}/g, '(?<$1>\\w+)')
    );
    
    for (const asset of assets) {
      const match = asset.name.match(regex);
      if (!match || !match.groups) continue;
      
      // Find placeholder with matching groups
      const placeholder = placeholders.find(p => {
        const pGroups = this.parsePlaceholder(p.placeholder);
        return Object.values(match.groups!).every(v => 
          pGroups.includes(v.toLowerCase())
        );
      });
      
      if (placeholder) {
        mappings.set(placeholder.placeholder, asset);
      }
    }
    
    return mappings;
  }
  
  private static parsePlaceholder(placeholder: string): string[] {
    // {{WORKCARD_DAY_01}} → ["workcard", "day", "01"]
    return placeholder
      .replace(/[{}]/g, '')
      .toLowerCase()
      .split(/[_\s-]+/)
      .filter(Boolean);
  }
}

// exportEngine.ts (extended)

export class ExportEngine {
  
  static async exportWithProfile(
    manifest: AssemblyManifest,
    assembledContent: string,
    assets: ProcessedAsset[],
    profileId: string
  ): Promise<ExportResult> {
    
    const profile = EXPORT_PROFILES.find(p => p.id === profileId);
    if (!profile) throw new Error('Profile not found');
    
    // Apply profile transformations
    const transformed = await this.applyProfileTransforms(
      assembledContent,
      assets,
      profile
    );
    
    // Export based on format
    switch (profile.format) {
      case 'html':
        return this.exportHTML(manifest, transformed, assets);
      case 'epub':
        return this.exportEPUB(manifest, transformed, assets, profile.config);
      case 'pdf':
        return this.exportPDF(manifest, transformed, profile.config);
    }
  }
  
  private static async applyProfileTransforms(
    content: string,
    assets: ProcessedAsset[],
    profile: ExportProfile
  ): Promise<string> {
    
    let transformed = content;
    
    // Font size adjustment
    if (profile.config.fontSize) {
      transformed = transformed.replace(
        /<body/,
        `<body style="font-size: ${profile.config.fontSize}px; line-height: ${profile.config.lineHeight}"`
      );
    }
    
    // Color mode conversion
    if (profile.config.colorMode === 'grayscale') {
      transformed = transformed.replace(
        /<style>/,
        '<style>* { filter: grayscale(100%); }'
      );
    }
    
    // Image optimization
    if (profile.config.optimizeImages) {
      for (const asset of assets.filter(a => a.type === 'raster')) {
        const optimized = await this.optimizeImage(asset, profile);
        transformed = transformed.replaceAll(asset.processedURL, optimized.processedURL);
      }
    }
    
    // Accessibility enhancements
    if (profile.config.accessibility) {
      transformed = this.addAccessibilityAttributes(transformed);
    }
    
    return transformed;
  }
  
  private static addAccessibilityAttributes(html: string): string {
    // Add ARIA labels, semantic HTML, etc.
    return html
      .replace(/<img/g, '<img role="img"')
      .replace(/<button/g, '<button role="button"')
      .replace(/<input/g, '<input aria-required="false"');
  }
}

static async listSnapshots(manifestTitle: string): Promise<ProjectSnapshot[]> {
const keys = await window.storage.list(this.STORAGE_PREFIX);
if (!keys) return [];

const snapshots = await Promise.all(
  keys.keys.map(key => this.loadSnapshot(key.replace(this.STORAGE_PREFIX, '')))

  .filter((s): s is ProjectSnapshot => s !== null)
  .filter(s => s.manifest.metadata.title === manifestTitle)
  .sort((a, b) => new Date(b.timestamp).getTime() - new Date(a.timestamp).getTime());

```typescript
// services/monitoring/performanceMonitor.ts

export class PerformanceMonitor {
  
  private static metrics: Map<string, number[]> = new Map();
  
  static measure<T>(operation: string, fn: () => T | Promise<T>): T | Promise<T> {
    const start = performance.now();
    
    const result = fn();
    
    if (result instanceof Promise) {
      return result.finally(() => {
        this.recordMetric(operation, performance.now() - start);
      }) as Promise<T>;
    } else {
      this.recordMetric(operation, performance.now() - start);
      return result;
    }
  }
  
  private static recordMetric(operation: string, duration: number) {
    if (!this.metrics.has(operation)) {
      this.metrics.set(operation, []);
    }
    this.metrics.get(operation)!.push(duration);
    
    // Keep last 100 measurements
    if (this.metrics.get(operation)!.length > 100) {
      this.metrics.get(operation)!.shift();
    }
  }
  
  static getStats(operation: string) {
    const measurements = this.metrics.get(operation) || [];
    if (measurements.length === 0) return null;
    
    const sorted = [...measurements].sort((a, b) => a - b);
    return {
      min: sorted[0],
      max: sorted[sorted.length - 1],
      avg: measurements.reduce((a, b) => a + b) / measurements.length,
      p50: sorted[Math.floor(sorted.length * 0.5)],
      p95: sorted[Math.floor(sorted.length * 0.95)],
      count: measurements.length
    };
  }
  
  static getAllStats() {
    return Array.from(this.metrics.keys()).map(op => ({
      operation: op,
      ...this.getStats(op)
    }));
  }
}
```

```
## 5. SECURITY & SANDBOXING

### 5.1 XSS Prevention Strategy

**Threat model:**
User wstrzykuje złośliwy HTML → XSS → kradzież danych / session hijacking.

**Defense layers:**
```
USER HTML INPUT
    ↓
[1] FORBIDDEN PATTERN CHECK
    - Regex dla: javascript:, on*, <script>, eval()
    - Reject natychmiast
    ↓
[2] DOMPURIFY SANITIZATION
    - Whitelist allowed tags/attrs
    - Remove wszystkie event handlers
    - Encode dangerous chars
    ↓
[3] SANDBOX WRAPPER
    - CSS isolation: isolation: isolate
    - Contain: layout style paint
    - No script execution context
    ↓
[4] CONTENT SECURITY POLICY (CSP)
    - script-src 'self'
    - object-src 'none'
    - frame-ancestors 'none'
    ↓
SAFE RENDERING
5.2 Allowed vs Forbidden
Allowed tags:

typescript
['div', 'span', 'p', 'h1-h6', 'ul', 'ol', 'li', 
 'table', 'tr', 'td', 'th', 'input', 'textarea', 
 'label', 'button', 'select', 'img', 'audio', 'video']
Forbidden tags:

typescript
['script', 'iframe', 'object', 'embed', 'applet', 
 'link', 'style', 'meta', 'base']
Forbidden attributes:

typescript
['onclick', 'onerror', 'onload', 'onmouseover', 
 'onfocus', 'onblur', ...]

Dla max security, wrap w Shadow DOM:

html
<div class="shadow-host">
  <template shadowrootmode="open">
    <style>
      :host { all: initial; display: block; }
      /* Scoped CSS */
    </style>

  </template>

CSS nie leakuje na zewnątrz
External CSS nie wpływa na workcard
JS nie może dostać się do parent DOM

## 7. PERFORMANCE GUIDELINES

### 7.1 Performance Targets

| Operation | Target | Method |
|-----------|--------|--------|
| Parse manifest | <100ms | Sync JSON.parse |
| Validate schema | <200ms | AJV validator |
| Process 1 image | <500ms | OffscreenCanvas |
| Process 1 HTML | <300ms | DOMPurify |
| Map 50 assets | <2s | Async parallel |
| Validate (local) | <500ms | Sync checks |
| Validate (AI) | <5s | Single API call |
| Inject placeholders | <3s | String replacement batch |
| Export HTML | <1s | Template + bundle |
| Export EPUB | <5s | Worker thread |
| Export PDF | <15s | Worker thread |

### 7.2 Optimization Strategies

**1. Lazy Loading Assets**
```typescript
// Don't load all assets upfront
const asset = mappings.get(placeholder);
if (!asset.loaded) {
  await loadAsset(asset); // Load on-demand
}
```

**2. Batch Processing**
```typescript
// Process images in parallel
await Promise.all(
  images.map(img => AssetProcessor.process(img, 'raster', config))
);
```

**3. Worker Threads dla Heavy Operations**
```typescript
// PDF generation off main thread
const worker = new Worker('/workers/pdf.worker.js');
worker.postMessage({ content, config });
```

**4. Incremental Rendering**
```typescript
// Don't wait for all sections, render progressively
sections.forEach((section, i) => {
  setTimeout(() => renderSection(section), i * 100);
});
```

**5. Memoization**
```typescript
const processedCache = new Map<string, ProcessedAsset>();

async function getProcessed(file: File): Promise<ProcessedAsset> {
  const key = `${file.name}_${file.size}`;
  if (processedCache.has(key)) {
    return processedCache.get(key)!;
  }
  const processed = await AssetProcessor.process(file);
  processedCache.set(key, processed);
  return processed;
}
```

### 7.3 Memory Management

**Problem:** 50+ high-res images = memory leak.

**Solution:**
```typescript
// Revoke object URLs after use
function cleanup(assets: ProcessedAsset[]) {
  assets.forEach(asset => {
    if (asset.processedURL.startsWith('blob:')) {
      URL.revokeObjectURL(asset.processedURL);
    }
  });
}

// Call after export
await ExportEngine.exportPDF(...);
cleanup(processedAssets);
```

## 10. TROUBLESHOOTING

### 10.1 Common Issues

**Problem: Manifest validation fails**
Error: Invalid manifest: data.sections[0].chapter should be >= 1

**Solution:** Check chapter numbering starts at 1, is sequential.

---

**Problem: Asset not mapping**
Warning: Placeholder {{WORKCARD_01}} not resolved

**Solution:** 
1. Check filename matches manifest path exactly
2. Use batch auto-map
3. Verify file uploaded successfully

---

**Problem: HTML injection fails / blank output**
Error: DOMPurify removed all content

**Solution:**
1. Check HTML doesn't contain `<script>` tags
2. Remove inline event handlers (`onclick=...`)
3. Use allowed tags only

---

**Problem: PDF export hangs**
Worker timeout after 30s

**Solution:**
1. Reduce image count/size
2. Split into multiple PDFs
3. Use "draft" quality profile

---

**Problem: Memory leak on large projects**
Out of memory error after processing 100+ images

**Solution:**
```typescript
// Revoke blob URLs periodically
setInterval(() => {
  processedAssets.forEach(a => {
    if (!a.inUse) URL.revokeObjectURL(a.processedURL);
  });
}, 60000); // Every minute
```

```typescript
import DOMPurify from 'dompurify';

export class HTMLSanitizer {
  private static FORBIDDEN_PATTERNS = [
    /javascript:/gi,
    /on\w+=/gi,
    /<script/gi,
    /<iframe/gi,
    /eval\(/gi
  ];
  
  static sanitize(html: string, strict: boolean = true): string {
    // LAYER 1: Pattern check
    for (const pattern of this.FORBIDDEN_PATTERNS) {
      if (pattern.test(html)) {
        throw new AssemblyError('SECURITY_VIOLATION', 'ERROR', 
          `Forbidden pattern: ${pattern.source}`);
      }
    }
    
    // LAYER 2: DOMPurify
    const clean = DOMPurify.sanitize(html, {
      ALLOWED_TAGS: this.ALLOWED_TAGS,
      ALLOWED_ATTR: this.ALLOWED_ATTRS,
      FORBID_TAGS: ['script', 'iframe', 'object'],
      FORBID_ATTR: ['onerror', 'onload', 'onclick']
    });
    
    // LAYER 3: Sandbox wrapper
    return `
      <div class="workcard-sandbox" style="isolation: isolate;">
        ${clean}
      </div>
    `;
  }
}
```

## C. ASSET PROCESSOR (assetProcessor.ts)
```typescript
export class AssetProcessor {
  static async process(
    file: File,
    type: AssetType,
    config: ProcessingConfig
  ): Promise<ProcessedAsset> {
    
    switch (type) {
      case 'workcard-html':
        return this.processHTML(file, config);
      case 'raster':
        return this.processImage(file, config);
      case 'audio':
        return this.processAudio(file, config);
    }
  }
  
  private static async processImage(file: File, config: ProcessingConfig) {
    // 1. Create ImageBitmap
    // 2. Resize via OffscreenCanvas
    // 3. Convert to WebP
    // 4. Generate checksum
    // 5. Return ProcessedAsset
  }
  
  private static async generateChecksum(blob: Blob): Promise<string> {
    const buffer = await blob.arrayBuffer();
    const hashBuffer = await crypto.subtle.digest('SHA-256', buffer);
    return Array.from(new Uint8Array(hashBuffer))
      .map(b => b.toString(16).padStart(2, '0'))
      .join('');
  }
}
```

```typescript
export class ExportEngine {
  static async exportHTML(
    manifest: AssemblyManifest,
    assembledContent: string,
    assets: ProcessedAsset[]
  ): Promise<ExportResult> {
    
    const html = `
<!DOCTYPE html>
<html lang="${manifest.metadata.language}">
<head>
  <meta charset="UTF-8">
  <title>${manifest.metadata.title}</title>
  <style>${await this.bundleCSS(manifest.assets.css)}</style>
</head>

  <script>${await this.bundleScripts(manifest.assets.scripts)}</script>
</body>
</html>
    `;

    const url = URL.createObjectURL(blob);
    
    return {
      success: true,
      format: 'html',
      size: blob.size,
      downloadURL: url,
      metadata: { assets: assets.length, generatedAt: new Date().toISOString() }
    };
  }
  
  static async exportPDF(...) {
    // Use Web Worker for heavy processing
    const worker = new Worker(new URL('../workers/pdf.worker.ts', import.meta.url));
    return new Promise((resolve, reject) => {
      worker.postMessage({ manifest, content, config });
      worker.onmessage = (e) => {
        if (e.data.success) resolve(e.data);
        else reject(e.data.error);
        worker.terminate();
      };
    });
  }
```

## STORAGE STRATEGY:

**NIGDY NIE UŻYWAJ localStorage/sessionStorage** (nie działa w Claude Artifacts)

**ZAMIAST TEGO:**
```typescript
// services/shared/storageService.ts

export class StorageService {
  
  // Save manifest
  static async saveManifest(projectId: string, manifest: AssemblyManifest) {
    await window.storage.set(
      `manifest:${projectId}`,
      JSON.stringify(manifest),
      false // personal data
    );
  }
  
  // Save asset mappings (split into chunks if > 5MB)
  static async saveMappings(projectId: string, mappings: Map<string, ProcessedAsset>) {
    const entries = Array.from(mappings.entries());
    const serialized = JSON.stringify(entries);
    
    if (serialized.length > 5_000_000) {
      // Split into chunks
      const chunkSize = 4_000_000;
      for (let i = 0; i < serialized.length; i += chunkSize) {
        const chunk = serialized.slice(i, i + chunkSize);
        await window.storage.set(
          `mappings:${projectId}:chunk${i / chunkSize}`,
          chunk,
          false
        );
      }
    } else {
      await window.storage.set(
        `mappings:${projectId}`,
        serialized,
        false
      );
    }
  }
  
  // Load project
  static async loadProject(projectId: string): Promise<{
    manifest: AssemblyManifest;
    mappings: Map<string, ProcessedAsset>;
  } | null> {
    try {
      const manifestResult = await window.storage.get(`manifest:${projectId}`);
      if (!manifestResult) return null;
      
      // Load mappings (check for chunks)
      let mappingsData = '';
      try {
        const direct = await window.storage.get(`mappings:${projectId}`);
        mappingsData = direct ? direct.value : '';
      } catch {
        // Load chunks
        let chunkIndex = 0;
        while (true) {
          try {
            const chunk = await window.storage.get(`mappings:${projectId}:chunk${chunkIndex}`);
            if (!chunk) break;
            mappingsData += chunk.value;
            chunkIndex++;
          } catch {
            break;
          }
        }
      }
      
      return {
        manifest: JSON.parse(manifestResult.value),
        mappings: new Map(JSON.parse(mappingsData))
      };
    } catch {
      return null;
    }
  }
}
```

                    <div className="mt-8 bg-purple-50 rounded-xl p-6 border-2 border-purple-200">
                      <h4 className="font-bold text-lg mb-4">❓ {lesson.quiz.title}</h4>
                      {lesson.quiz.questions.map((q, qIdx) => (
                        <div key={q.id} className="mb-6">
                          <p className="font-bold mb-3">{qIdx + 1}. {q.question}</p>
                          <div className="space-y-2">
                            {q.options.map((opt, oIdx) => (
                              <label key={oIdx} className="flex items-center gap-3 p-3 bg-white rounded-lg cursor-pointer hover:bg-purple-100 transition-all">
                                <input type="radio" name={`q-${qIdx}`} />
                                <span>{opt}</span>
                              </label>
                            ))}
                          </div>
                        </div>
                      ))}
                    </div>
                  )}
                </div>
              ))}
            </div>
          ))}
        </div>
      </div>
    );
  }

  return null;
};

const LessonEditor: React.FC<{
  lesson: Lesson;
  onChange: (lesson: Lesson) => void;
}> = ({ lesson, onChange }) => {
  const imageInputRef = useRef<HTMLInputElement>(null);
  const audioInputRef = useRef<HTMLInputElement>(null);

  const handleImageUpload = (e: React.ChangeEvent<HTMLInputElement>) => {
    const files = e.target.files;
    if (!files) return;
    
    Array.from(files).forEach(file => {
      const url = URL.createObjectURL(file);
      const newImage: MediaFile = {
        id: Date.now().toString() + Math.random(),
        name: file.name,
        url,
        type: file.type
      };
      
      onChange({
        ...lesson,
        multimedia: {
          ...lesson.multimedia,
          images: [...(lesson.multimedia.images || []), newImage]
        }
      });
    });
  };

  const handleAudioUpload = (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (!file) return;
    
    const url = URL.createObjectURL(file);
    const newAudio: MediaFile = {
      id: Date.now().toString(),
      name: file.name,
      url,
      type: file.type
    };
    
    onChange({
      ...lesson,
      multimedia: { ...lesson.multimedia, audio: newAudio }
    });
  };

  return (
    <div className="max-w-4xl mx-auto space-y-8">
      {/* BASIC INFO */}
      <div className="bg-white rounded-2xl p-8 shadow-lg border border-slate-200">
        <h2 className="text-xs font-black uppercase text-slate-400 mb-4 tracking-wider">
          Podstawowe Informacje
        </h2>
        <div className="space-y-4">
          <input
            type="text"
            placeholder="Tytuł lekcji"
            value={lesson.title}
            onChange={(e) => onChange({ ...lesson, title: e.target.value })}
            className="w-full px-4 py-3 border border-slate-200 rounded-xl font-bold text-xl focus:outline-none focus:border-blue-500"
          />
          <input
            type="text"
            placeholder="Podtytuł (opcjonalnie)"
            value={lesson.subtitle}
            onChange={(e) => onChange({ ...lesson, subtitle: e.target.value })}
            className="w-full px-4 py-3 border border-slate-200 rounded-xl focus:outline-none focus:border-blue-500"
          />
          <textarea
            placeholder="Treść lekcji..."
            value={lesson.content}
            onChange={(e) => onChange({ ...lesson, content: e.target.value })}
            className="w-full px-4 py-3 border border-slate-200 rounded-xl h-64 resize-none focus:outline-none focus:border-blue-500"
          />
        </div>
      </div>

      {/* MULTIMEDIA */}
      <div className="bg-white rounded-2xl p-8 shadow-lg border border-slate-200">
        <h2 className="text-xs font-black uppercase text-slate-400 mb-4 tracking-wider">
          Multimedia
        </h2>
        
        <div className="space-y-6">
          {/* IMAGES */}
          <div>
            <div className="flex justify-between items-center mb-3">
              <label className="font-bold text-sm">Obrazy</label>
              <button
                onClick={() => imageInputRef.current?.click()}
                className="px-4 py-2 bg-blue-500 text-white rounded-lg text-sm font-bold flex items-center gap-2"
              >
                <Image size={16} />
                Dodaj Obraz
              </button>
              <input
                ref={imageInputRef}
                type="file"
                accept="image/*"
                multiple
                className="hidden"
                onChange={handleImageUpload}
              />
            </div>
            
            {lesson.multimedia.images && lesson.multimedia.images.length > 0 && (
              <div className="grid grid-cols-3 gap-4">
                {lesson.multimedia.images.map(img => (
                  <div key={img.id} className="relative group">
                    <img src={img.url} alt={img.name} className="w-full h-32 object-cover rounded-lg" />
                    <button
                      onClick={() => {
                        onChange({
                          ...lesson,
                          multimedia: {
                            ...lesson.multimedia,
                            images: lesson.multimedia.images?.filter(i => i.id !== img.id)
                          }
                        });
                      }}
                      className="absolute top-2 right-2 p-2 bg-red-500 text-white rounded-lg opacity-0 group-hover:opacity-100 transition-opacity"
                    >
                      <Trash2 size={16} />
                    </button>
                  </div>
                ))}
              </div>
            )}
          </div>

          {/* AUDIO */}
          <div>
            <div className="flex justify-between items-center mb-3">
              <label className="font-bold text-sm">Audio</label>
              <button
                onClick={() => audioInputRef.current?.click()}
                className="px-4 py-2 bg-green-500 text-white rounded-lg text-sm font-bold flex items-center gap-2"
              >
                <Music size={16} />
                {lesson.multimedia.audio ? 'Zmień Audio' : 'Dodaj Audio'}
              </button>
              <input
                ref={audioInputRef}
                type="file"
                accept="audio/*"
                className="hidden"
                onChange={handleAudioUpload}
              />
            </div>
            
            {lesson.multimedia.audio && (
              <div className="bg-green-50 p-4 rounded-xl flex justify-between items-center">
                <div>
                  <p className="font-bold text-sm mb-2">🎧 {lesson.multimedia.audio.name}</p>
                  <audio controls className="w-full" src={lesson.multimedia.audio.url} />
                </div>
                <button
                  onClick={() => {
                    onChange({
                      ...lesson,
                      multimedia: { ...lesson.multimedia, audio: undefined }
                    });
                  }}
                  className="p-2 bg-red-500 text-white rounded-lg"
                >
                  <Trash2 size={16} />
                </button>
              </div>
            )}
          </div>

          {/* VIDEO */}
          <div>
            <label className="font-bold text-sm block mb-3">Video YouTube</label>
            <input
              type="text"
              placeholder="ID wideo YouTube (np. dQw4w9WgXcQ)"
              value={lesson.multimedia.video || ''}
              onChange={(e) => onChange({
                ...lesson,
                multimedia: { ...lesson.multimedia, video: e.target.value }
              })}
              className="w-full px-4 py-3 border rounded-xl focus:outline-none focus:border-blue-500"
            />
            {lesson.multimedia.video && (
              <div className="mt-4">
                <iframe
                  width="100%"
                  height="300"
                  src={`https://www.youtube.com/embed/${lesson.multimedia.video}`}
                  className="rounded-xl"
                  allowFullScreen
                />
              </div>
            )}
          </div>
        </div>
      </div>

      {/* WORKSHEET */}
      <div className="bg-white rounded-2xl p-8 shadow-lg border border-slate-200">
        <div className="flex justify-between items-center mb-4">
          <h2 className="text-xs font-black uppercase text-slate-400 tracking-wider">
            Karta Pracy
          </h2>
          {!lesson.worksheet && (
            <button
              onClick={() => onChange({
                ...lesson,
                worksheet: {
                  title: 'Karta Pracy',
                  questions: [{ id: '1', question: '', type: 'textarea' }]
                }
              })}
              className="px-4 py-2 bg-amber-500 text-white rounded-lg text-sm font-bold"
            >
              + Dodaj Kartę
            </button>
          )}
        </div>
        
        {lesson.worksheet && (
          <div className="space-y-4">
            <input
              type="text"
              value={lesson.worksheet.title}
              onChange={(e) => onChange({
                ...lesson,
                worksheet: { ...lesson.worksheet!, title: e.target.value }
              })}
              className="w-full px-4 py-2 border rounded-lg font-bold"
            />
            
            {lesson.worksheet.questions.map((q, idx) => (
              <div key={q.id} className="p-4 bg-slate-50 rounded-xl">
                <div className="flex gap-2 mb-2">
                  <input
                    type="text"
                    placeholder="Pytanie..."
                    value={q.question}
                    onChange={(e) => {
                      const newQuestions = [...lesson.worksheet!.questions];
                      newQuestions[idx].question = e.target.value;
                      onChange({
                        ...lesson,
                        worksheet: { ...lesson.worksheet!, questions: newQuestions }
                      });
                    }}
                    className="flex-1 px-3 py-2 border rounded-lg"
                  />
                  <button
                    onClick={() => {
                      const newQuestions = lesson.worksheet!.questions.filter((_, i) => i !== idx);
                      onChange({
                        ...lesson,
                        worksheet: { ...lesson.worksheet!, questions: newQuestions }
                      });
                    }}
                    className="p-2 bg-red-500 text-white rounded-lg"
                  >
                    <Trash2 size={16} />
                  </button>
                </div>
                <select
                  value={q.type}
                  onChange={(e) => {
                    const newQuestions = [...lesson.worksheet!.questions];
                    newQuestions[idx].type = e.target.value as 'text' | 'textarea';
                    onChange({
                      ...lesson,
                      worksheet: { ...lesson.worksheet!, questions: newQuestions }
                    });
                  }}
                  className="px-3 py-2 border rounded-lg text-sm"
                >
                  <option value="text">Krótka odpowiedź</option>
                  <option value="textarea">Długa odpowiedź</option>
                </select>
              </div>
            ))}
            
            <button
              onClick={() => {
                const newQuestions = [
                  ...lesson.worksheet!.questions,
                  { id: Date.now().toString(), question: '', type: 'textarea' as const }
                ];
                onChange({
                  ...lesson,
                  worksheet: { ...lesson.worksheet!, questions: newQuestions }
                });
              }}
              className="w-full py-2 border-2 border-dashed border-slate-300 rounded-lg text-sm text-slate-500 hover:border-blue-500"
            >
              + Dodaj Pytanie
            </button>
          </div>
        )}
      </div>

      {/* QUIZ */}
      <div className="bg-white rounded-2xl p-8 shadow-lg border border-slate-200">
        <div className="flex justify-between items-center mb-4">

            Quiz
          </h2>
          {!lesson.quiz && (
            <button
              onClick={() => onChange({
                ...lesson,
                quiz: {
                  title: 'Quiz',
                  questions: [{
                    id: '1',
                    question: '',
                    options: ['Opcja A', 'Opcja B', 'Opcja C'],
                    correctAnswer: 0
                  }]
                }
              })}
              className="px-4 py-2 bg-purple-500 text-white rounded-lg text-sm font-bold"
            >
              + Dodaj Quiz
            </button>

        </div>
        
        {lesson.quiz && (
          <div className="space-y-4">
            <input
              type="text"
              value={lesson.quiz.title}
              onChange={(e) => onChange({
                ...lesson,
                quiz: { ...lesson.quiz!, title: e.target.value }
              })}
              className="w-full px-4 py-2 border rounded-lg font-bold"
            />
            
            {lesson.quiz.questions.map((q, qIdx) => (
              <div key={q.id} className="p-4 bg-purple-50 rounded-xl">
                <div className="flex gap-2 mb-3">
                  <input
                    type="text"
                    placeholder="Pytanie..."
                    value={q.question}
                    onChange={(e) => {
                      const newQuestions = [...lesson.quiz!.questions];
                      newQuestions[qIdx].question = e.target.value;
                      onChange({
                        ...lesson,
                        quiz: { ...lesson.quiz!, questions: newQuestions }
                      });
                    }}
                    className="flex-1 px-3 py-2 border rounded-lg font-bold"
                  />
                  <button
                    onClick={() => {
                      const newQuestions = lesson.quiz!.questions.filter((_, i) => i !== qIdx);
                      onChange({
                        ...lesson,
                        quiz: { ...lesson.quiz!, questions: newQuestions }
                      });
                    }}
                    className="p-2 bg-red-500 text-white rounded-lg"
                  >
                    <Trash2 size={16} />
                  </button>
                </div>
                
                <div className="space-y-2">
                  {q.options.map((opt, oIdx) => (
                    <div key={oIdx} className="flex gap-2">
                      <input
                        type="radio"
                        name={`correct-${qIdx}`}
                        checked={q.correctAnswer === oIdx}
                        onChange={() => {
                          const newQuestions = [...lesson.quiz!.questions];
                          newQuestions[qIdx].correctAnswer = oIdx;
                          onChange({
                            ...lesson,
                            quiz: { ...lesson.quiz!, questions: newQuestions }
                          });
                        }}
                        className="mt-3"
                      />
                      <input
                        type="text"
                        value={opt}
                        onChange={(e) => {
                          const newQuestions = [...lesson.quiz!.questions];
                          newQuestions[qIdx].options[oIdx] = e.target.value;
                          onChange({
                            ...lesson,
                            quiz: { ...lesson.quiz!, questions: newQuestions }
                          });
                        }}
                        className="flex-1 px-3 py-2 border rounded-lg"
                      />
                    </div>
                  ))}
                  <button
                    onClick={() => {
                      const newQuestions = [...lesson.quiz!.questions];
                      newQuestions[qIdx].options.push(`Opcja ${String.fromCharCode(65 + q.options.length)}`);
                      onChange({
                        ...lesson,
                        quiz: { ...lesson.quiz!, questions: newQuestions }
                      });
                    }}
                    className="w-full py-2 border border-dashed rounded-lg text-sm text-purple-600"
                  >
                    + Dodaj Opcję
                  </button>
                </div>
              </div>
            ))}
            
            <button
              onClick={() => {
                const newQuestions = [
                  ...lesson.quiz!.questions,
                  {
                    id: Date.now().toString(),
                    question: '',
                    options: ['Opcja A', 'Opcja B', 'Opcja C'],
                    correctAnswer: 0
                  }
                ];
                onChange({
                  ...lesson,
                  quiz: { ...lesson.quiz!, questions: newQuestions }
                });
              }}
              className="w-full py-2 border-2 border-dashed border-purple-300 rounded-lg text-sm text-purple-600 hover:border-purple-500"
            >
              + Dodaj Pytanie
            </button>
          </div>
        )}
      </div>
    </div>
  );
};

function generateWebbook(webbook: Webbook): string {
  const modulesHTML = webbook.modules.map((module, mIdx) => 
    module.lessons.map((lesson, lIdx) => `
      <div class="page ${mIdx === 0 && lIdx === 0 ? 'active' : ''}" id="page-${mIdx}-${lIdx}">
        <div class="page-header">
          <h1>${lesson.title}</h1>
          ${lesson.subtitle ? `<p class="subtitle">${lesson.subtitle}</p>` : ''}
        </div>
        
        <div class="content">
          <p style="white-space: pre-wrap; line-height: 1.8;">${lesson.content}</p>
        </div>
        
        ${lesson.multimedia.images && lesson.multimedia.images.length > 0 ? `
        <div class="media-gallery">
          ${lesson.multimedia.images.map(img => `
            <img src="${img.url}" alt="${img.name}" />
          `).join('')}
        </div>
        ` : ''}

        <div class="audio-player">
          <p class="audio-title">🎧 ${lesson.multimedia.audio.name}</p>

        <div class="video-container">
          <iframe src="https://www.youtube.com/embed/${lesson.multimedia.video}" allowfullscreen></iframe>
        </div>
        ` : ''}
        
        ${lesson.worksheet ? `
        <div class="worksheet">
          <h3>${lesson.worksheet.title}</h3>
          ${lesson.worksheet.questions.map(q => `
            <div class="form-group">
              <label>${q.question}</label>

                ? '<textarea rows="4"></textarea>'

                <input
                  ref={projectInputRef}
                  type="file"
                  accept=".json"
                  className="hidden"

            {showTemplates && (

                        </div>
                      )}
                    </div>
                  ))}
                </div>
              </div>
            </div>
          </aside>

          <main className="flex-1 p-8 overflow-y-auto h-[calc(100vh-73px)]">

                  <p className="text-slate-400">lub dodaj nowy moduł/lekcję</p>
                </div>
              </div>
            )}

      </div>
    );
  }

  // ============================================================================
  // PREVIEW
  // ============================================================================
  
  if (currentView === 'preview') {
    return (
      <div className="min-h-screen bg-white">
        <header className="border-b border-slate-200 bg-white sticky top-0 z-50">
          <div className="px-8 py-4 flex justify-between items-center">
            <button
              onClick={() => setCurrentView('builder')}
              className="px-4 py-2 bg-slate-100 hover:bg-slate-200 rounded-lg font-bold flex items-center gap-2"
            >
              <ChevronRight size={20} className="rotate-180" />
              Powrót
            </button>
            <h2 className="text-lg font-bold text-slate-600">Podgląd Webbooka</h2>
          </div>
        </header>
        
        <div className="max-w-4xl mx-auto px-8 py-16">
          <div className="mb-16 text-center border-b-4 border-slate-900 pb-12">
            <h1 className="text-6xl font-black mb-6">{webbook.metadata.title}</h1>
            <p className  
 <p className="text-2xl text-slate-600">{webbook.metadata.subtitle}</p>
            {webbook.metadata.author && (
              <p className="text-xl text-slate-500 mt-4">Autor: {webbook.metadata.author}</p>
            )}
          </div>
          
          {webbook.modules.map((module, mIdx) => (
            <div key={module.id} className="mb-16">

                {module.title}
              </h2>
              
              {module.lessons.map((lesson, lIdx) => (

                    <div className="grid grid-cols-2 gap-4 mb-6">
                      {lesson.multimedia.images.map(img => (
                        <img key={img.id} src={img.url} alt={img.name} className="w-full rounded-xl" />
                      ))}
                    </div>
                  )}
                  
                  {lesson.multimedia.audio && (
                    <div className="bg-blue-50 p-6 rounded-xl mb-6">
                      <p className="font-bold mb-3">🎧 {lesson.multimedia.audio.name}</p>
                      <audio controls className="w-full" src={lesson.multimedia.audio.url} />
                    </div>
                  )}
                  
                  {lesson.multimedia.video && (
                    <div className="mb-6">
                      <iframe
                        width="100%"
                        height="400"
                        src={`https://www.youtube.com/embed/${lesson.multimedia.video}`}
                        className="rounded-xl"
                        allowFullScreen
                      />
                    </div>
                  )}
                  
                  {lesson.worksheet && (
                    <div className="mt-8 bg-white rounded-xl p-6 border-2 border-slate-200">
                      <h4 className="font-bold text-lg mb-4">📋 {lesson.worksheet.title}</h4>
                      {lesson.worksheet.questions.map(q => (
                        <div key={q.id} className="mb-4">
                          <label className="block font-bold mb-2">{q.question}</label>
                          {q.type === 'textarea' ? (
                            <textarea className="w-full p-3 border rounded-lg" rows={4} />
                          ) : (
                            <input type="text" className="w-full p-3 border rounded-lg" />
                          )}
                        </div>
                      ))}
                    </div>
                  )}
                  
                  {lesson.quiz && (
                    <div className="mt-8 bg-purple-50 rounded-xl p-6 border-2 border-purple-200">
                      <h4 className="font-bold text-lg mb-4">❓ {lesson.quiz.title}</h4>
                      {lesson.quiz.questions.map((q, qIdx) => (
                        <div key={q.id} className="mb-6">
                          <p className="font-bold mb-3">{qIdx + 1}. {q.question}</p>
                          <div className="space-y-2">
                            {q.options.map((opt, oIdx) => (
                              <label key={oIdx} className="flex items-center gap-3 p-3 bg-white rounded-lg cursor-pointer hover:bg-purple-100 transition-all">
                                <input type="radio" name={`q-${qIdx}`} />
                                <span>{opt}</span>
                              </label>
                            ))}
                          </div>
                        </div>
                      ))}
                    </div>
                  )}
                </div>
              ))}
            </div>
          ))}
        </div>
      </div>
    );
  }

  return null;
};

// WEBBOOK GENERATOR
// ============================================================================

function generateWebbook(webbook: Webbook): string {
  const modulesHTML = webbook.modules.map((module, mIdx) => 
    module.lessons.map((lesson, lIdx) => `
      <div class="page ${mIdx === 0 && lIdx === 0 ? 'active' : ''}" id="page-${mIdx}-${lIdx}">
        <div class="page-header">
          <h1>${lesson.title}</h1>
          ${lesson.subtitle ? `<p class="subtitle">${lesson.subtitle}</p>` : ''}
        </div>
        
        <div class="content">
          <p style="white-space: pre-wrap; line-height: 1.8;">${lesson.content}</p>
        </div>
        
        ${lesson.multimedia.images && lesson.multimedia.images.length > 0 ? `
        <div class="media-gallery">
          ${lesson.multimedia.images.map(img => `
            <img src="${img.url}" alt="${img.name}" />
          `).join('')}
        </div>
        ` : ''}
        
        ${lesson.multimedia.audio ? `
        <div class="audio-player">
          <p class="audio-title">🎧 ${lesson.multimedia.audio.name}</p>
          <audio controls src="${lesson.multimedia.audio.url}"></audio>
        </div>
        ` : ''}
        
        ${lesson.multimedia.video ? `
        <div class="video-container">
          <iframe src="https://www.youtube.com/embed/${lesson.multimedia.video}" allowfullscreen></iframe>
        </div>
        ` : ''}
        
        ${lesson.worksheet ? `
        <div class="worksheet">
          <h3>${lesson.worksheet.title}</h3>
          ${lesson.worksheet.questions.map(q => `
            <div class="form-group">
              <label>${q.question}</label>
              ${q.type === 'textarea' 
                ? '<textarea rows="4"></textarea>' 
                : '<input type="text" />'
              }
            </div>
          `).join('')}
          <button class="btn btn-save" onclick="alert('Odpowiedzi zapisane!')">Zapisz odpowiedzi</button>
        </div>
        ` : ''}
        
        ${lesson.quiz ? `
        <div class="quiz">
          <h3>${lesson.quiz.title}</h3>
          ${lesson.quiz.questions.map((q, qIdx) => `
            <div class="quiz-question">
              <p class="question-text">${qIdx + 1}. ${q.question}</p>
              <div class="quiz-options">
                ${q.options.map((opt, oIdx) => `
                  <label class="quiz-option">
                    <input type="radio" name="q${mIdx}-${lIdx}-${qIdx}" value="${oIdx}" />
                    <span>${opt}</span>
                  </label>
                `).join('')}
              </div>
            </div>
          `).join('')}
          <button class="btn btn-primary" onclick="alert('Quiz ukończony!')">Sprawdź odpowiedzi</button>
        </div>
        ` : ''}
        
        <div class="navigation-buttons">
          ${lIdx > 0 || mIdx > 0 ? `<button class="btn btn-secondary" onclick="showPage('page-${lIdx > 0 ? mIdx : mIdx - 1}-${lIdx > 0 ? lIdx - 1 : webbook.modules[mIdx - 1]?.lessons.length - 1 || 0}')">← Poprzednia</button>` : ''}
          ${lIdx < module.lessons.length - 1 || mIdx < webbook.modules.length - 1 ? `<button class="btn btn-primary" onclick="showPage('page-${lIdx < module.lessons.length - 1 ? mIdx : mIdx + 1}-${lIdx < module.lessons.length - 1 ? lIdx + 1 : 0}')">Następna →</button>` : ''}
        </div>
      </div>
    `).join('')
  ).join('');

  const sidebarHTML = webbook.modules.map((module, mIdx) => `
    <div class="sidebar-section">
      <div class="sidebar-title">${module.title}</div>
      ${module.lessons.map((lesson, lIdx) => `
        <a href="#" class="sidebar-link ${mIdx === 0 && lIdx === 0 ? 'active' : ''}" 
           onclick="showPage('page-${mIdx}-${lIdx}'); return false;">
          ${lesson.title}
        </a>
      `).join('')}
    </div>
  `).join('');

  return `<!DOCTYPE html>
<html lang="${webbook.metadata.language}">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">

      // Hide all pages
      document.querySelectorAll('.page').forEach(page => {
        page.classList.remove('active');
      });

      const page = document.getElementById(pageId);
      if (page) {

        document.querySelectorAll('.sidebar-link').forEach(link => {
          link.classList.remove('active');

        const activeLink = document.querySelector(\`[onclick="showPage('\${pageId}'); return false;"]\`);
        if (activeLink) {
